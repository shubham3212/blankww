        JMenuItem aboutItem = new JMenuItem("About");
        helpMenu.add(aboutItem);

        // Add menus to the JMenuBar
        menuBar.add(fileMenu);
        menuBar.add(helpMenu);

        // Set JMenuBar for the frame
        frame.setJMenuBar(menuBar);

        // Create checkboxes panel
        JCheckBox cppCheckbox = new JCheckBox("CPP");
        JCheckBox javaCheckbox = new JCheckBox("Java");

        JPanel checkboxesPanel = new JPanel(new GridLayout(1, 2)); // GridLayout for side-by-side checkboxes
        checkboxesPanel.add(cppCheckbox);
        checkboxesPanel.add(javaCheckbox);

        // Create JTextArea with JScrollPane
        JTextArea textArea = new JTextArea();
        JScrollPane scrollPane = new JScrollPane(textArea);

        // Create main panel to hold checkboxes panel and text area
        JPanel mainPanel = new JPanel(new BorderLayout());

        // Create a bigger panel to center within the main panel
        JPanel centerPanel = new JPanel(new BorderLayout());
        centerPanel.add(checkboxesPanel, BorderLayout.WEST); // Place checkboxes on the left
        centerPanel.add(scrollPane, BorderLayout.CENTER); // Centered position for text area

        mainPanel.add(centerPanel, BorderLayout.CENTER); // Center the bigger panel within the main panel

        // Create bottom panel with radio buttons
        JPanel bottomPanel = new JPanel();
        JRadioButton oneYearRadioButton = new JRadioButton("1 Year");
        JRadioButton twoYearsRadioButton = new JRadioButton("2 Years");
        JRadioButton threeYearsRadioButton = new JRadioButton("3 Years");
        JRadioButton fourYearsRadioButton = new JRadioButton("4 Years");

        ButtonGroup buttonGroup = new ButtonGroup();

        buttonGroup.add(twoYearsRadioButton);
        buttonGroup.add(threeYearsRadioButton);

        bottomPanel.add(oneYearRadioButton);
        bottomPanel.add(twoYearsRadioButton);
        bottomPanel.add(threeYearsRadioButton);
        bottomPanel.add(fourYearsRadioButton);

        mainPanel.add(bottomPanel, BorderLayout.SOUTH); // Add the bottom panel to the main panel

            import java.sql.Connection;
            import java.sql.DriverManager;
            import java.sql.ResultSet;
            import java.sql.SQLException;
            import java.sql.Statement;
            
            public class CreateDatabaseTable {
                public static void main(String[] args) {
                    Connection connection = null;
                    try {
                        // Open a connection
                        Class.forName("com.mysql.cj.jdbc.Driver");
                        connection = DriverManager.getConnection("jdbc:mysql://localhost:3306/mybookstore", "root", "root");
            
                        // Create a statement
                        Statement stmt = connection.createStatement();
            
                        // Select data from the 'books' table
                        String strSelect = "SELECT title, price, qty FROM books";
                        System.out.print("The SQL statement is " + strSelect + "\n");
            
                        ResultSet rset = stmt.executeQuery(strSelect);
                        System.out.print("The records selected are: ");
            
                        int rowCount = 0;
                        while (rset.next()) {
                            String title = rset.getString("title");
                            double price = rset.getDouble("price");
                            int qty = rset.getInt("qty");
                            System.out.println(title + " " + price + ", " + qty);
                            rowCount++;
                        }
            
                        System.out.println("Total number of records - " + rowCount);
            
                    } catch (ClassNotFoundException | SQLException e) {
                        e.printStackTrace();
                    } finally {
                        try {
                            // Close resources
                            if (connection != null) {
                                connection.close();
                            }
                        } catch (SQLException e) {
                            e.printStackTrace();
                        }
                    }
                }
            }

            package socketprogram;

import java.io.DataInputStream;
import java.io.DataOutputStream;
import java.io.IOException;
import java.net.ServerSocket;
import java.net.Socket;
import java.net.SocketTimeoutException;

public class GreetingServer extends Thread {
    private ServerSocket serverSocket;

    public GreetingServer(int port) throws IOException {
        serverSocket = new ServerSocket(port);
        serverSocket.setSoTimeout(1000000);
    }

    public void run() {
        while (true) {
            try {
                System.out.println("Waiting for client on port " + serverSocket.getLocalPort() + "...");
                Socket server = serverSocket.accept();
                System.out.println("Just connected to " + server.getRemoteSocketAddress());

                DataInputStream in = new DataInputStream(server.getInputStream());
                DataOutputStream out = new DataOutputStream(server.getOutputStream());
                out.writeUTF("Thanks for connecting to " + server.getLocalSocketAddress() + "\nGoodbye!");
                server.close();
            } catch (SocketTimeoutException e) {
                System.out.println("Socket timed out");
                break;
            } catch (IOException e) {
                e.printStackTrace();
                break;
            }
        }
    }

    public static void main(String[] args) {
        int port = Integer.parseInt(args[0]);
        try {
            new GreetingServer(port).start();
        } catch (IOException e) {
            e.printStackTrace();
        }
    }
}

--Addition Interfac-------------------------------------------------------------------------------------------------------------
            import java.rmi.Remote;
import java.rmi.RemoteException;

public interface AdditionInterface extends Remote {
    int add(int a, int b) throws RemoteException;
}

--Addition Implementaion---------------------------------------------------------------------------------------------------
import java.rmi.RemoteException;
import java.rmi.server.UnicastRemoteObject;

public class Addition extends UnicastRemoteObject implements AdditionInterface {
    public Addition() throws RemoteException {
        super();
    }

    public int add(int a, int b) throws RemoteException {
        return a + b;
    }
}

--Addition Server---------------------------------------------------------------------------------------------------------------
import java.rmi.Naming;
import java.rmi.registry.LocateRegistry;

public class AdditionServer {
    public static void main(String[] args) {
        try {
            LocateRegistry.createRegistry(1099); // Create the RMI registry on port 1099
            Addition hello = new Addition();
            Naming.rebind("rmi://localhost/ABC", hello);
            System.out.println("Addition server is ready");
        } catch (Exception e) {
            System.out.println("Addition server failed: " + e);
        }
    }
}

--Addition Client---------------------------------------------------------------------------------------------------------------
import java.rmi.Naming;

public class AdditionClient {
    public static void main(String[] args) {
        AdditionInterface hello;
        try {
            hello = (AdditionInterface) Naming.lookup("rmi://localhost/ABC");
            int result = hello.add(9, 10);
            System.out.println("Result is " + result);
        } catch (Exception e) {
            System.out.println("Hello client exception: " + e);
        }
    }
}
